129 void step() {
130 if (enabled)
131 {
132
133 heating = tempControl.stateIsHeating();
134 cooling = tempControl.stateIsCooling();
135 doorOpen = PSensor(tempControl.door)->sense();
136 // with no serial and no calculation here we get 1500-2000x speedup
137 // with this code enabled, around 1300x speedup
138 // with serial, drops to 300x speedup
139 double fermDiff = beerFerment();
140 double heatingDiff = chamberHeating();
141 double coolingDiff = chamberCooling();
142 double doorDiff = doorLosses();
143
144 double newBeerTemp = beerTemp + fermDiff;
145 double newFridgeTemp = fridgeTemp + heatingDiff + coolingDiff + doorDiff;
146
147 TempPair beerTx;
148 TempPair roomTx;
149 currentRoomTemp = roomTemp();
150 chamberBeerTransfer(fridgeTemp, beerTemp, beerTx);
151 chamberRoomTransfer(fridgeTemp, currentRoomTemp, roomTx);
152
153 newFridgeTemp += beerTx.t1 + roomTx.t1; // transfer from beer and environment
154 newBeerTemp += beerTx.t2;
155
156 fridgeTemp = newFridgeTemp;
157 beerTemp = newBeerTemp;
158
159 time += 1;
160 }
161 updateSensors();
162 }

254 double roomTemp()
255 {
256 if (minRoomTemp==maxRoomTemp)
257 return minRoomTemp;
258
259 unsigned long secondsInADay = 60*60*24UL;
260 double p = (double(time%secondsInADay)/double(secondsInADay))*(TWO_PI);
261 double s = sin(p);
262 double mid = (minRoomTemp+maxRoomTemp)/2;
263 double half = mid-minRoomTemp;
264 double r = mid + s*half;
265 return r;
266 }

312 /* Compute the heat transferred between two heat energy potentials. Returns the temperature change
313 of each potential. */
314 void heatTransfer(const HeatPotential& p1, const HeatPotential& p2, double k, TempPair& result)
315 {
316 double e1 = (p2.temp-p1.temp)*k; // the energy transferred - from p2 to p1
317 result.t1 = e1/p1.capacity; // change in temperature for the increase in energy
318 result.t2 = -e1/p2.capacity;
319 }

333 double beerFerment() {
334 double days = hours()/24.0;
335 double scale = 0;
336 if (days>5)
337 scale = 0;
338 else if (days>2)
339 scale = 1.0-((days-3)/3);
340 else if (days>1)
341 scale = 1.0;
342 double power = scale*fermentPowerMax;
343 return power / beerHeatCapacity;
344 }
