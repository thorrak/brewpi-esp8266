97inline bool isAssignable(DeviceType type, DeviceHardware hardware)
98{
99 return (hardware==DEVICE_HARDWARE_PIN && (type==DEVICETYPE_SWITCH_ACTUATOR || type==DEVICETYPE_SWITCH_SENSOR))
100#if BREWPI_DS2413
101 || (hardware==DEVICE_HARDWARE_ONEWIRE_2413 && (type==DEVICETYPE_SWITCH_ACTUATOR || (DS2413_SUPPORT_SENSE && type==DEVICETYPE_SWITCH_SENSOR)))
102#endif
103#ifdef HAS_BLUETOOTH
104 || (hardware==DEVICE_HARDWARE_BLUETOOTH_INKBIRD && type==DEVICETYPE_TEMP_SENSOR)
105 || (hardware==DEVICE_HARDWARE_BLUETOOTH_TILT && type==DEVICETYPE_TEMP_SENSOR)
106#endif
107#ifdef EXTERN_SENSOR_ACTUATOR_SUPPORT
108 || (hardware==DEVICE_HARDWARE_TPLINK_SWITCH && type==DEVICETYPE_SWITCH_ACTUATOR)
109#endif
110 || (hardware==DEVICE_HARDWARE_ONEWIRE_TEMP && type==DEVICETYPE_TEMP_SENSOR)
111 || (hardware==DEVICE_HARDWARE_NONE && type==DEVICETYPE_NONE);
112}

269 int8_t enumerateActuatorPins(uint8_t offset)
270 {
271#if BREWPI_ACTUATOR_PINS && defined(ARDUINO)
272#if BREWPI_STATIC_CONFIG<=BREWPI_SHIELD_REV_A
273 switch (offset) {
274 case 0: return heatingPin;
275 case 1: return coolingPin;
276 default:
277 return -1;
278 }
279#elif BREWPI_STATIC_CONFIG>=BREWPI_SHIELD_REV_C
280 switch (offset) {
281 case 0: return actuatorPin1;
282 case 1: return actuatorPin2;
283 case 2: return actuatorPin3;
284 case 3: return actuatorPin4;
285 default: return -1;
286 }
287#endif
288#endif
289 return -1;
290 }

292 int8_t enumerateSensorPins(uint8_t offset) {
293#if BREWPI_SENSOR_PINS && defined(ARDUINO)
294 if (offset==0)
295 return doorPin;
296#endif
297 return -1;
298 }

303 int8_t enumOneWirePins(uint8_t offset) {
304#ifdef ARDUINO
305#ifdef oneWirePin
306 if (offset == 0)
307 return oneWirePin;
308#elif defined(beerSensorPin) && defined(fridgeSensorPin)
309 if (offset==0)
310 return beerSensorPin;
311 if (offset==1)
312 return fridgeSensorPin;
313#endif
314#endif
315 return -1;
316 }
