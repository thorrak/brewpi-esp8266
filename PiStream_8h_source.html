71 int readPersistent(const int timeout = 10) {
72 uint8_t retries = 0;
73 while (available() == 0) {
74 // Uses delay as delayMicroseconds doesn't yield like delay does
75 delay(1);
76 yield();
77 retries++;
78 if (retries >= timeout) {
79 return -1;
80 }
81 }
82 return stream.read();
83 };

104 void print(const char *out) {

106 stream.print(out);

108 for(uint16_t x = 0; x < strlen(out) ; x++) {
109 if((unsigned)(intBuffOn + 1) < Config::PiLink::intBufferSize()) {
110 intBuff[intBuffOn] = out[x];

204 void sendJsonMessage(const char prefix, const JsonDocument &doc) {
205 if (prefix) {
206 print(prefix);
207 print(':');
208 }
209
210 if(Config::PiLink::bufferPrints) {
211 serializeJson(doc, stream);
212 } else {
213 char buf[2048];
214 serializeJson(doc, buf, 2048);
215 print(buf);
216 }
217 printNewLine();
218 }

231 void sendSingleItemJsonMessage(const char prefix, JsonDocument &doc) {
232 // Pull the device object from the document array and promote it to be root
233 DynamicJsonDocument shallowDoc(doc.capacity());
234
235 for (auto kvp : doc.as<JsonArray>()[0].as<JsonObject>()) {
236 shallowDoc[kvp.key()] = kvp.value();
237 }
238
239 sendJsonMessage(prefix, shallowDoc);
240 }
